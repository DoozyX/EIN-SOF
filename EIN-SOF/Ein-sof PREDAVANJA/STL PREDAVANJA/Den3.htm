4 Експлицитна специјализација

 

4.1 Експлицитна специјализација на function templates

 

Понекогаш е пожелно да се изврши заобиколуање на кодот генериран со template. Тоа може да се постигне со преклопување со конвенционална функција, но постои механизам да се изврши преку експлицитна специјализација на template. Во следниот пример е прикажана специјализација со повеќе различни синтакси.

 

//

// Demonsstracija na eksplicitna specijalizacija kaj function template

 

 

#include <iostream>

#include <string>

 

using namespace std;

 

template<class T>

size_t bytes(T& t)

{

      cout << "(koristam primaren template)\n";

      return sizeof t;

}

 

size_t bytes(char*& s)

{

      cout<<"(koristam preklopuvanje char* funkcija)\n";

      return strlen(s)+1; // za string sakame da bide vratena negovata dolzina

}

 

template <>

size_t bytes<>(string& s)

{

      cout<<"(koristam eksplicitna string specijalizacija)\n";

      return sizeof s;

}

 

template <>

size_t bytes<float>(float& x)

{

      cout<<"(koristam eksplicitna float specijalizacija)\n";

      return sizeof x;

}

 

main()

{

      int i;

      cout << "broj na bajti vo i "<<bytes(i)<<endl;

      char *s="Zdravo";

      cout << "broj na bajti vo s "<<bytes(s)<<endl;

      string t;

      cout << "broj na bajti vo t "<<bytes(t)<<endl;

      float x;

      cout << "broj na bajti vo x "<<bytes(x)<<endl;

      double d;

      cout << "broj na bajti vo d "<<bytes(d)<<endl;

 

}

 

Излезот е:

 

(koristam primaren template)

broj na bajti vo i 4

(koristam preklopuvanje char* funkcija)

broj na bajti vo s 7

(koristam eksplicitna string specijalizacija)

broj na bajti vo t 16

(koristam eksplicitna float specijalizacija)

broj na bajti vo x 4

(koristam primaren template)

broj na bajti vo d 8

 

Следниот пример илустрира ситуација во која преклопување на кодот во function template –от е неопходно:

#include <iostream>

using namespace std ;

 

//max vrakja maksimum od dva elementi od tip T, kade za  T e

// definiran operatorot >

 

template <class T>

T max(T a, T b)

{

    return a > b ? a : b ;

}

 

int main()

{   

    cout << "max(10, 15) = " << max(10, 15) << endl ;

    cout << "max('k', 's') = " << max('k', 's') << endl ;

    cout << "max(10.1, 15.2) = " << max(10.1, 15.2) << endl ;

    cout << "max(\"Aladdin\", \"Jasmine\") = " << max("Aladdin", "Jasmine") << endl;//vrsi sporeduvanje na adresi

    return 0 ;

}

 
Излезот е:
 

max(10, 15) = 15

max('k', 's') = s

max(10.1, 15.2) = 15.2

max("Aladdin", "Jasmine") = Aladdin

 

Овој резултат не е точен. Повикот на max("Aladdin", "Jasmine") предизвикува генерирање на код за max(char*, char*), која всушност не споредува стрингови, туку нивните адреси. Затоа, за да се кореира однесувањето на max за одредени типови, може да се користат специјализации. Претходниот пример може да се преработи на следниот начин:

 

#include <iostream>

#include <cstring>

using namespace std ;

 

//max vrakja maksimum od dva

template <class T>

T max(T a, T b)

{

    return a > b ? a : b ;

}

 

// Speciaјlizacija на max за char*

template <>

char* max(char* a, char* b)

{

    return strcmp(a, b) > 0 ? a : b ;

}

 

int main()

{

   

    cout << "max(10, 15) = " << max(10, 15) << endl ;

    cout << "max('k', 's') = " << max('k', 's') << endl ;

    cout << "max(10.1, 15.2) = " << max(10.1, 15.2) << endl ;

    cout << "max(\"Aladdin\", \"Jasmine\") = " << max("Aladdin", "Jasmine") << endl ; //Dava tocen rezultat

    return 0 ;

}

 
Излезот е:
 

max(10, 15) = 15

max('k', 's') = s

max(10.1, 15.2) = 15.2

max("Aladdin", "Jasmine") = Jasmine

 

4.2 Експлицитна специјализација на class templates

 

Може да се врши и експлицитна специјализација на class template-и. Следната програма го илустрира тоа:

 

 

//

//Demonsstracija na eksplicitna specijalizacija kaj class template

 

 

#include <iostream>

 

using namespace std;

 

template<typename T>

class A

{

public:

      A(){cout << "Primaren template\n";}

};

template <>

class A<char>

{

public:

      A(){cout << "char specijalizacija\n";}

};

 

template <>

class A<float>

{

public:

      A(){cout << "float specijalizacija\n";}

};

 

 

main()

{

      A<int> a1;

      A<char> a2;

      A<float> a3;

      A<double> a4;

 

}

 

Излезот е:

 

Primaren template

char specijalizacija

float specijalizacija

Primaren template

Class template-ите можат да бидат и парцијално специјализирани, т.е. некои од типовите во листата формални типови можат да се специфицираат, а некои да се остават празни. При специјализација се избира оној template чија листа на аргументи најмногу се поклопува со онаа во специјализацијата, слично како кај преклопување на функции. Следниот пример илустрира разни случаи на парцијална специјализација кај class template :

// Vrsi parcijalna specijalizacija

 

#include <iostream>

using namespace std;

 

 

template<class T, class U> class C {

public:

  void f() { cout << "Primaren template\n"; }

};

 

template<class U> class C<int, U> {

public:

  void f() { cout << "T == int\n"; }

};

 

template<class T> class C<T, double> {

public:

  void f() { cout << "U == double\n”; }

};

 

template<class T, class U> class C<T*, U> {

public:

  void f() { cout << "T*  \n”; }

};

 

template<class T, class U> class C<T, U*> {

public:

  void f() { cout << "U*  \n”; }

};

 

template<class T, class U> class C<T*, U*> {

public:

  void f() { cout << "T* i U*  \n”; }

};

 

template<class T> class C<T, T> {

public:

  void f() { cout << "T == U\n”; }

};

 

int main() {

  C<float, int>().f();    // 1: Primaren template

  C<int, float>().f();    // 2: T == int

  C<float, double>().f(); // 3: U == double

  C<float, float>().f();  // 4: T == U

  C<float*, float>().f(); // 5: T* [T e float]

  C<float, float*>().f(); // 6: U* [U e float]

  C<float*, int*>().f();  // 7: T* i U* [float,int]

  // Slednite specijalizacii se dvosmisleni.

//   8: C<int, int>().f();

//   9: C<double, double>().f();

//  10: C<float*, float*>().f();

//  11: C<int, int*>().f();

//  12: C<int*, int*>().f();

} // kraj na main

 

Параметрите можат да се специфицираат парцијално, во зависност дали се покажувачи или се еднакви. На пример, 8 е двосмислено бидејќи одговара и на class C<int, U> и на class C<U, U>.

Напомена: Visual C++ 5.0 и Visual C++ 6.0 не поддржува парцијална специјализација на класи. Претходниот пример ќе ја предизвика следната грешка: compiler error C2989: template class has already been defined as a non-template class.

 

5 Статички членови и променливи

Секоја template класа или класа генерирана од template има своја копија на сите статички променливи или членки. Секоја инстанца на function template има своја копија на сите статички променливи дефинирани во рамки на функцијата. На пример, во: 

 

//

//Vklucuvanje na staticki clenovi

 

#include <iostream>

using namespace std ;

 

template <class T>

class X

{

    public:

static T s ;

} ;

 

int main()

{

 

X<int> xi ;

      X<char*> xc ;

}

 

X<int> има статичка компонента s од тип int, а X<char*> има статичка компонента s од тип char*.

Секоја инстанца (специјализација) на class template има своја копија од статичката ченка.Следниот пример прикажува доделување на вредности на статички членови:

//

//Demonstracija na staticki clenovi

 

#include <iostream>

using namespace std ;

 

template <class T>

class X

{

    public:

      static T s ;

} ;

 

template <class T> T X<T>::s = 0 ;

template <> int X<int>::s = 3 ; //int specijalizacija

template <> char* X<char*>::s = "Hello" ;//char* specijalizacija

 

int main()

{

      X<int> xi ;

      cout << "xi.s = " << xi.s << endl ;

 

       X<char*> xc ;

      cout << "xc.s = " << xc.s << endl ;

     

      return 0 ;

}

Излез од програмата:

xi.s = 10

xc.s = Hello

Секоја инстанца (специјализација) на function template има своја копија од статичката ченка. На пример,

//

//Vklucuvanje na staticki clenovi kaj function template

 

#include <iostream>

using namespace std ;

 

template <class T>

void f(T t)

{

      static T s ;

      s = t ;

      cout << "s = " << s << endl ;

}

 

int main()

{

      f(10) ;

      f("Hello") ;

     

      return 0 ;

}

 

ќе даде излез:

 

s = 10

s = Hello

Тука f<int>(int) има статичка компонента s од тип int, а f<char*>(char*) има статичка компонента s од тип char*.

6 Template-и и пријателски функции

Пријателски односи можат да се воспостават помеѓу class template и глобална функција, функција членка на друга класа (која може да биде и специјализација на template) и класа (која исто така може да биде и специјализација на template). 

Ако во рамките на class template деклариран како:

template <class T> class X

се декларира:

friend void f1();

функцијата f1 станува пријателска за секоја специјализација инстанцирана од неа.

Декларацијата

friend void f2( X<T> &);

за тип T, ако Т е  float, ја прави функцијата f2(X<float>&) пријателска само за специјализацијата X<float>.

Во рамки на class template може да се декларира функција членка на некоја класа да биде пријателска за сите негови специјализации. Се користи името на таа класа и операторот за припадност на класа ::.

Ако во рамките на class template деклариран како:

template <class T> class X

се декларира:

friend void А::f4();

членката f4 на класата А станува пријателска за сите специјализации на template –от.

За class template деклариран како:

template <class T> class X

декларацијата:

friend void C<T>::f5(X<T>&);

за тип T, ако Т е  float, ја прави функцијата членка

C<float>::f5(X<float>&)

пријателска само за специјализацијата X<float>.

За class template деклариран како:

template <class T> class X

декларацијата:

friend class Y;

ги прави пријателски сите функции членки на Y за сите специјализации на X.

За class template деклариран како:

template <class T> class X

декларацијата:

friend class Z<T>;

ги прави пријателски сите функции членки на специјализацијата Z<T> за специјализацијата на X<T>. На пример, сите членки на Z<float> се пријателски за X<float>.

 

#include <iostream.h>

#include <conio.h>

 

template <class T>

class B;

 

 

template <class T>

class A {

 T val;

   friend void swap (A<T> &, B<T> &);

 public:

   A(){cout <<"val="; cin >> val; cout << endl;}

   void disp() {cout << "val=" << val << endl;}

};

 

template <class T>

class B {

 T x;

 

   friend void swap (A<T> &, B<T> &);

 public:

   B(){cout <<"x="; cin >> x; cout << endl;}

   void disp() {cout << "x=" << x  << endl;}

};

 

template <class T>

void swap( A<T> &p, B<T> &q)

{

   T buf;

   buf=p.val;p.val=q.x;q.x=buf;

}

main()

{

 

  A<int> objA;

  B<int> objB;

 

  swap<int>(objA, objB);

  objA.disp();

  objB.disp();

  return 0;

}

 

 

Излез од програмата:

 

val=1

 

x=2

 

val=2

x=1

 

Вежби:

 

 

1.      Напиши class template Zapis<T> кој содржи три приватни членки: x, y и z од тип Т. Креирај function template kontrolor кој ќе проверува дали збирот на елементите x, y и z во Zapis надминува 10000. Напиши соодветна програма за демонстрација со инстанцирање на објекти од тип Zapis<int> и Zapis<double>. 

2.      Креирај class template Mnozestvo<T> во со функции членки за додавање, прикажување на елементи на множеството и определување на најголем елемент во множеството. Како податочна структура за сместување на елементите на множеството може да се користи поле. Секој елемент може да се појави само еднаш. Секое множество содржи static компонента во која е сместен бројот на елементи на множеството. Креирај три множества: А од int, B од double и C од char*. Користи соодветна специјализација за коректно определување на најголем елемент во множеството од стрингови. Програмата треба да имплементира мени со следните опции:

2.      Додај елемент во B

3.      Додај елемент во C

4.      Прикажи го А

5.      Прикажи го B

6.      Прикажи го C

7.      Прикажи најголеми елементи

8.      Прикажи број на елементи во множествата
